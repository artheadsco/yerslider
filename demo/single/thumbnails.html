var myslider = new YerSlider();
myslider.init({
	sliderid: '.mysliderclass',
	slidegap: 10,
	slidegroupresp: {
		0: 1,
		450: 2,
		900: 3
	},
	loop: 'rollback',
	animationspeed: 1000,
	bullets: true,
	autoplay: true,
	autoplayinterval: 1000,
	autoplaydelaystart: 0,
	autoplaystoponhover: true
	
thumbs: true,
	thumbstemplates: {
		'1': {
			'html': '<p>{{thumb-text}}</p>',
			'class': 'thumb-template-1'
		}
	},
	thumbsclickable: true,
	thumbsscript: function( p ) {
		var thumbsscript = new YerSliderThumbs();
		thumbsscript.init({
			obj: p.obj
		});
	}


});
